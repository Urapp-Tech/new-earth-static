   document.addEventListener('DOMContentLoaded', () => {
      const mainTitle = document.querySelector('.main-title');
      const heroTrees = document.querySelector('.hero-trees');
      const heroBuilding = document.querySelector('.hero-building');

      let scrollStep = 0; // Keeps track of animation steps

      const showElement = (element) => element.classList.add('visible');
      const hideElement = (element) => element.classList.remove('visible');

      const handleScroll = (direction) => {
        if (direction === 'down') {
          if (scrollStep === 0) {
            // Step 1: Show main title
            showElement(mainTitle);
            scrollStep++;
          } else if (scrollStep === 1) {
            // Step 2: Show trees and building
            showElement(heroTrees);
            showElement(heroBuilding);
            scrollStep++;
          }
        } else if (direction === 'up') {
          if (scrollStep === 2) {
            // Step 2: Hide trees and building
            hideElement(heroTrees);
            hideElement(heroBuilding);
            scrollStep--;
          } else if (scrollStep === 1) {
            // Step 1: Hide main title
            hideElement(mainTitle);
            scrollStep--;
          }
        }
      };

      let lastScrollY = 0;

      // Listen for wheel events (desktop)
      window.addEventListener('wheel', (event) => {
        const direction = event.deltaY > 0 ? 'down' : 'up';
        handleScroll(direction);
      });

      // Optional: Touch events (mobile)
      let touchStartY = 0;

      window.addEventListener('touchstart', (event) => {
        touchStartY = event.touches[0].clientY;
      });

      window.addEventListener('touchmove', (event) => {
        const touchEndY = event.touches[0].clientY;
        const direction = touchStartY - touchEndY > 0 ? 'down' : 'up';
        handleScroll(direction);
      });
    });
